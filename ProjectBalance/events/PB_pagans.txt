namespace = meneth
#Ensure Pagans have enough Piety to hire Holy Orders
character_event = {
	id = meneth.100
	desc = "meneth.100.desc"
	picture = "GFX_evt_battle"
	border = "GFX_event_normal_frame_religion"
	
	only_rulers = yes
	
	trigger = {
		always = no #Disabled for now
		religion_group = pagan_group
		NOT = { primary_title = { tier = baron }}
		OR = {
			NOT = { piety = 75 }
			AND = {
				NOT = { piety = 125 }
				culture = pecheneg
			}
		}
		ai = yes
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "meneth.100.a"
		ai_chance = { factor = 100 }
		piety = 15
	}
}

# Ilkhanate converts to Reformed Tengri
character_event = {
	id = meneth.101
	title = "EVTNAME60010"
	desc = meneth.101.desc
	picture = "GFX_evt_mongols"
	border = "GFX_event_normal_frame_war"
	major = yes
	
	only_playable = yes
	
	trigger = {
		primary_title = {
			title = e_il-khanate
			holder_scope = {
				religion = tengri_pagan
				NOT = { trait = zealous }
			}
		}
		had_global_flag = { flag = il_khanate_arrival days = 14600 }
		NOT = { had_global_flag = { flag = il_khanate_arrival days = 29200 } }
		NOT = { has_global_flag = il_khanate_converts }
		any_realm_province = {
			religion = tengri_pagan_reformed
		}
	}

	mean_time_to_happen = {
		months = 210
		modifier = {
			factor = 0.75
			any_demesne_province = {
				is_capital = yes
				religion = tengri_pagan_reformed
			}
		}
	}
	
	immediate = {
		set_global_flag = il_khanate_converts
	}
	
	option = {	# Reformed Tengri
		name = "EVTOPTA60010"
		trigger = {
			has_landed_title = e_il-khanate
		}
		religion = tengri_pagan_reformed
		capital_scope = {
			religion = tengri_pagan_reformed
		}
		any_realm_character = {
			limit = { religion = tengri_pagan }
			religion = tengri_pagan_reformed
			if = {
				limit = {
					is_playable = yes
					capital_scope = {
						religion = tengri_pagan
					}
				}
				capital_scope = {
					religion = tengri_pagan_reformed
				}
			}
		}
	}
	# Other rulers are informed
	option = {
		name = "EVTOPTB60010"
		trigger = {
			religion_group = muslim
			NOT = { has_landed_title = e_il-khanate }
		}
	}
	option = {
		name = "EVTOPTC60010"
		trigger = {
			religion_group = christian
			NOT = { has_landed_title = e_il-khanate }
		}
	}
	option = {
		name = "EVTOPTD60010"
		trigger = {
			OR = {
				religion_group = pagan_group
				religion_group = zoroastrian_group
			}
			NOT = { religion = tengri_pagan_reformed }
			NOT = { has_landed_title = e_il-khanate }
		}
	}
	option = {
		name = "EXCELLENT"
		trigger = {
			religion = tengri_pagan_reformed
			NOT = { has_landed_title = e_il-khanate }
		}
	}
}
